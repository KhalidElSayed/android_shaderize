
precision mediump float;
varying vec2 vPosition;

void main() {
	gl_FragColor = vec4(0.0, 0.0, 0.5, 0.0);

	vec2 coord = vPosition;
	if (abs(coord.x) > 0.9) {
		float t = (abs(coord.x) - 0.9) * 10.0;
		float x = 0.5 + coord.x * 0.5;
		gl_FragColor = mix(gl_FragColor, vec4(x, 0.0, 0.0, 0.0), t);
	}
	/*
	if (abs(coord.y) > 0.9) {
		float t = (abs(coord.y) - 0.9) * 10.0;
		float y = 0.5 + coord.y * 0.5;
		gl_FragColor = mix(gl_FragColor, vec4(0.0, y, 0.0, 0.0), t);
	}
	*/
}
