
uniform mat4 uModelViewM;
uniform mat4 uProjM;
uniform mat4 uNormalM;
uniform float uSaturation;

attribute vec3 aPosition;
attribute vec3 aNormal;
attribute vec3 aColor;

varying vec4 vColor;

void main() {
	gl_Position = uProjM * (uModelViewM * vec4(aPosition, 1.0));
	vec4 normal = uNormalM * vec4(aNormal, 1.0);
	
	vColor = vec4(aColor * normal.z, 1.0);
	float gray = dot(vColor.rgb, vec3(0.3, 0.59, 0.11));
	vColor.rgb = mix(vColor.rgb, vec3(gray), uSaturation);
}
